Tệp src/chat.worker.js xử lý tải mô hình ngầm (Singleton Pattern) và phát luồng (stream) từng token về Main Thread. 
import { pipeline, TextStreamer } from "@huggingface/transformers";

class ChatbotPipelineSingleton {
    static task = "text-generation";
    static model = "onnx-community/Qwen2.5-0.5B-Instruct";
    static instance = null;

    static async getInstance(progress_callback = null) {
        if (this.instance === null) {
            this.instance = await pipeline(this.task, this.model, {
                device: "wasm", // Tương thích mọi trình duyệt
                dtype: "q4",    // Lượng tử hóa 4-bit (~350MB)
                progress_callback,
            });
        }
        return this.instance;
    }
}

self.addEventListener("message", async (event) => {
    const { messages } = event.data;
    try {
        const generator = await ChatbotPipelineSingleton.getInstance((progress) => {
            if (progress.status === "progress") {
                self.postMessage({ type: "DOWNLOAD_PROGRESS", percentage: progress.percentage });
            }
        });
        
        self.postMessage({ type: "START_GENERATION" });
        let responseText = "";
        
        const streamer = new TextStreamer(generator.tokenizer, {
            on_token_callback: (token) => {
                responseText += token;
                self.postMessage({ type: "STREAM_TOKEN", fullText: responseText });
            }
        });

        await generator(messages, { max_new_tokens: 256, streamer: streamer });
        self.postMessage({ type: "GENERATION_COMPLETE" });
    } catch (e) {
        self.postMessage({ type: "ERROR", error: e.message });
    }
});
Kết nối UI với Web Worker qua `useRef` và xử lý state thời gian thực. (Xem bản mô phỏng trực quan ở phần dưới). 
import React, { useState, useEffect, useRef } from 'react';

export default function GeneralChatbot() {
    const [messages, setMessages] = useState([]);
    const worker = useRef(null);

    useEffect(() => {
        worker.current = new Worker(new URL('./chat.worker.js', import.meta.url), { type: 'module' });
        
        worker.current.addEventListener('message', (e) => {
            const { type, percentage, fullText } = e.data;
            if (type === "DOWNLOAD_PROGRESS") setDownloadProgress(percentage);
            else if (type === "STREAM_TOKEN") {
                setMessages(prev => {
                    const updated = [...prev];
                    updated[updated.length - 1].content = fullText;
                    return updated;
                });
            }
        });
        return () => worker.current.terminate();
    }, []);

    // ... Hàm gửi tin nhắn (postMessage) & Render JSX
}
